
# Character classes

 As found in `<ctype.h>`.

   isalpha(c)    [a-zA-Z]
   isupper(c)    [A-Z]
   islower(c)    [a-z]
   isdigit(c)    [0-9]
   isalnum(c)    isalpha(c) | isdigit(c)
   isspace(c)    \t \f \v \n \r
   ispunct(c)    ! " # $ % & ' ( ) * + , - . / : ; < = > ? @ [ \ ] ^ _ ` { | } ~
   isxdigit(c)   [a-fA-F0-9]
   isgraph(c)
   isctrl(c)
   
   tolower(c)    return c converted to lower case
   toupper(c)    return c converted to upper case
